function startScanner() { if (!html5QrCode) { html5QrCode = new Html5Qrcode("reader"); } const config = { fps: 30, disableFlip: false, // ปิดการกลับภาพ (กล้องหลังไม่ต้องกลับซ้ายขวา) videoConstraints: { width: { ideal: 1920 }, height: { ideal: 1080 } } }; // เปลี่ยนวิธี: ขอรายชื่อกล้องทั้งหมดฮาร์ดแวร์ "ก่อน" เปิด Html5Qrcode.getCameras().then(devices => { if (devices && devices.length > 0) { // 1. พยายามหากล้องที่มีคำระบุว่าเป็นกล้องหลัง (back, rear, environment, หลัง) let backCamera = devices.find(c => c.label.toLowerCase().includes('back') || c.label.toLowerCase().includes('rear') || c.label.toLowerCase().includes('environment') || c.label.toLowerCase().includes('หลัง') ); // 2. ถ้าหาชื่อไม่เจอจริงๆ (มือถือบางรุ่นไม่ยอมตั้งชื่อเลนส์) // มือถือ 99% กล้องแรก (index 0) คือกล้องหน้า ให้เราบังคับเลือกกล้องตัวที่ 2 หรือตัวสุดท้ายแทน if (!backCamera) { backCamera = devices.length > 1 ? devices[devices.length - 1] : devices[0]; } // สั่งเปิดกล้องจาก Device ID ของกล้องหลังแบบเจาะจง! html5QrCode.start(backCamera.id, config, onScanSuccess, onScanFailure) .catch(e => { console.warn("เปิดจาก ID ไม่สำเร็จ กำลังลองวิธีที่ 2...", e); // Fallback 1: บังคับกล้องหลังด้วยคำสั่ง Exact html5QrCode.start({ facingMode: { exact: "environment" } }, config, onScanSuccess, onScanFailure) .catch(err2 => { // Fallback 2: วิธีทั่วไป html5QrCode.start({ facingMode: "environment" }, config, onScanSuccess, onScanFailure); }); }); } else { showAlert("ไม่พบกล้องในอุปกรณ์นี้", "error"); document.getElementById('qr-modal').style.display = "none"; } }).catch(err => { console.error("ดึงข้อมูลฮาร์ดแวร์ไม่ได้ จะลองบังคับเปิดกล้องหลังแบบ Exact...", err); // ถ้าระบบบล็อกไม่ให้อ่านรายชื่อกล้อง ให้บังคับเปิดกล้องหลังขั้นสูงสุด html5QrCode.start({ facingMode: { exact: "environment" } }, config, onScanSuccess, onScanFailure) .catch(e => { html5QrCode.start({ facingMode: "environment" }, config, onScanSuccess, onScanFailure) .catch(e2 => { showAlert("ไม่สามารถเข้าถึงกล้องได้ กรุณาตรวจสอบการตั้งค่า", "error"); document.getElementById('qr-modal').style.display = "none"; }); }); }); }
